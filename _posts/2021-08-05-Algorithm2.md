# [프린터](https://programmers.co.kr/learn/courses/30/lessons/42587)

```python
def solution(pri, loc):
    answer = 0
    while len(pri) > 0:
        if loc ==0:
            if pri[0] < max(pri):
                pri.append(pri.pop(0))
                loc = len(pri) - 1
            else:
                return answer + 1
        else:
            if pri[0] < max(pri):
                pri.append(pri.pop(0))
                loc -= 1
            else:
                pri.pop(0)
                loc -= 1
                answer += 1
```

## 다른 사람 풀이 1

```python
def solution(p, l):
    ans = 0
    m = max(p) #최댓값
    while True: #하나라도 있을 때까지 while문 동작
        v = p.pop(0) #첫번째 요소 추출
        if m == v: #첫번째 요소가 최댓값이라면
            ans += 1
            if l == 0:
                break
            else:
                l -= 1
            m = max(p) #추출한 나머지 list중에서 최댓값
        else:
            p.append(v) #list 마지막
            if l == 0: #location이 0이라면
                l = len(p)-1
            else:
                l -= 1
    return ans
```

## 다른 사람 풀이2

```python
def solution(priorities, location):
    queue =  [(i,p) for i,p in enumerate(priorities)]
    answer = 0
    while True:
        cur = queue.pop(0)
        if any(cur[1] < q[1] for q in queue): #하나라도 존재한다면(any)
            queue.append(cur)
        else:
            answer += 1
            if cur[0] == location:
                return answer
```
