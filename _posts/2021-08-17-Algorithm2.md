```python
def solution(scores):

    scores = list(map(list, zip(*scores)))

    answer = ""
    for index, score in enumerate(scores):
        if score[index] == min(score) or score[index] == max(score):
            if score.count(score[index]) == 1:
                del score[index]

        value = sum(score)/len(score)
        if value >= 90:
            answer += "A"
        elif 90 > value >= 80:
            answer += "B"
        elif 80 > value >= 70:
            answer += "C"
        elif 70 > value >= 50:
            answer += "D"
        else:
            answer += "F"

    return answer
```
