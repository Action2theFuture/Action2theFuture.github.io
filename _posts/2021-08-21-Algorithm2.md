def solution(citations):
    citations.sort()
    answer = []
    for i in range(1,len(citations)+1):
        N = 0
        n = 0
        for citation in citations:
            if i <= citation:
                N += 1
            elif i >= citation:
                n += 1

        if i <= N:
            answer.append(i)
    if not answer:
        return 0
    return max(answer)

def solution(citations):
    citations.sort(reverse=True)
    answer = max(map(min, enumerate(citations, start=1)))
    return answer
